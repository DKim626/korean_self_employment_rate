\section{KEAPS Self-employment Rate}\label{keaps-self-employment-rate}

This project analyzes \textbf{self-employment rates} using Korean
Economically Active Population Survey (KEAPS) \textbf{from 1981 to
2024}. It calculates and visualizes trends by \textbf{year,
industry(KSIC), age, and gender}.

\subsection{Overview}\label{overview}

\begin{itemize}
\tightlist
\item
  \textbf{Data source}: KEAPS (Korean Economically Active Population
  Survey)

  \begin{itemize}
  \tightlist
  \item
    Available at {[}MDIS{]}
    (\url{https://mdis.kostat.go.kr/eng/index.do})
  \end{itemize}
\item
  \textbf{Scope}

  \begin{itemize}
  \tightlist
  \item
    \textbf{Period}: 1981 - 2024 (monthly KEAPS microdata)
  \item
    \textbf{Unit of analysis}: Individuals of age 15+
  \item
    \textbf{Key variables}:

    \begin{itemize}
    \tightlist
    \item
      Employment status, industry (KSIC), gender \& age group
    \item
      Self-employment classification

      \begin{itemize}
      \tightlist
      \item
        \textbf{Type 1 (OECD)}: Family workers + self-employed with and
        without employees
      \item
        \textbf{Type 2 (KOSIS)}: Self-employed with and without
        employees
      \item
        \textbf{Type 3 (Project)}: Self-employed without employees only
      \end{itemize}
    \end{itemize}
  \item
    \textbf{Aggregation level}:

    \begin{itemize}
    \tightlist
    \item
      Yearly trends
    \item
      Industry-level trends
    \item
      Age \& Gender group trends
    \end{itemize}
  \end{itemize}
\item
  \textbf{Outputs}

  \begin{itemize}
  \tightlist
  \item
    \textbf{Monthly Outputs}

    \begin{itemize}
    \tightlist
    \item
      \texttt{month\_trends\_age\_group\_2.rdata}, etc.: monthly trends

      \begin{itemize}
      \tightlist
      \item
        \texttt{month\_overall}: monthly self-employment rates
      \item
        \texttt{month\_ind}: monthly rates by industry
      \item
        \texttt{month\_as}: monthly rates by age \& gender group
      \item
        \texttt{month\_as\_wg}: monthly rates within age \& gender group
      \item
        \texttt{month\_supp}: monthly labor participation \&
        unemployment rates
      \end{itemize}
    \end{itemize}
  \item
    \textbf{Yearly Outputs}

    \begin{itemize}
    \tightlist
    \item
      \texttt{year\_trends\_age\_group\_2.rdata}, etc.: yearly trends

      \begin{itemize}
      \tightlist
      \item
        \texttt{year\_overall}: yearly self-employment rates
      \item
        \texttt{year\_ind}: yearly rates by industry
      \item
        \texttt{year\_as}: yearly rates by age \& gender group
      \item
        \texttt{year\_as\_wg}: yearly rates within age \& gender group
      \item
        \texttt{year\_supp}: yearly labor participation \& unemployment
        rates
      \end{itemize}
    \end{itemize}
  \end{itemize}
\end{itemize}

\subsection{How to Run}\label{how-to-run}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Place raw \texttt{.csv} files in \texttt{data/raw/}
\item
  Place KOSIS \texttt{.R} scripts in \texttt{R/raw/}
\item
  In RStudio (project opened with \texttt{.Rproj}), open
  \texttt{main\_pipeline.R}
\item
  Run all sections
\item
  Outputs will be saved in \texttt{data/output/} as \texttt{.rdata}
  files
\item
  Images will be saved in \texttt{image/}
\end{enumerate}

\subsection{Analysis Pipeline}\label{analysis-pipeline}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Process raw data}

  \begin{itemize}
  \tightlist
  \item
    Raw \texttt{.csv} files are processed with KOSIS-provided scripts
  \end{itemize}
\item
  \textbf{Load raw data}

  \begin{itemize}
  \tightlist
  \item
    \texttt{.rds} files from 1981--2024 stored in
    \texttt{data/processed/}
  \end{itemize}
\item
  \textbf{Standardize variables}

  \begin{itemize}
  \tightlist
  \item
    Aligns consistent variable names across years
  \end{itemize}
\item
  \textbf{Recode variables}

  \begin{itemize}
  \tightlist
  \item
    Translates Korean -\textgreater{} English
  \item
    Groups ages hierarchically:

    \begin{itemize}
    \tightlist
    \item
      5-year intervals
    \item
      60+, 70+, 80+ categories
    \end{itemize}
  \item
    Classifies self-employment into 3 types (see above)
  \end{itemize}
\item
  \textbf{Calculate monthly rates}

  \begin{itemize}
  \tightlist
  \item
    Self-employment rate (overall, by industry, by/within group)
  \item
    Supplementary: labor force participation \& unemployment rate
  \end{itemize}
\item
  \textbf{Aggregate to yearly outputs}

  \begin{itemize}
  \tightlist
  \item
    Averages monthly results by year
  \end{itemize}
\end{enumerate}

\subsection{Known Issues}\label{known-issues}

\begin{itemize}
\tightlist
\item
  \textbf{KOSIS codes}: Use \texttt{"CP949"} to read \texttt{.R} scripts
  correctly

  \begin{itemize}
  \tightlist
  \item
    \emph{RStudio -\textgreater{} File -\textgreater{} Reopen with
    Encoding -\textgreater{} ``CP949''}
  \end{itemize}
\item
  \textbf{1983 survey}: Economic activity data highly likely corrupted

  \begin{itemize}
  \tightlist
  \item
    Participation rate is 1 in most months; results unreliable
  \end{itemize}
\item
  \textbf{1984 survey}: \texttt{weight} variable miscoded

  \begin{itemize}
  \tightlist
  \item
    Contains prefix ``\xb''; fixed manually
  \end{itemize}
\item
  \textbf{1986 survey}: \texttt{recode\_factor()} in original KOSIS
  script was misaligned

  \begin{itemize}
  \tightlist
  \item
    Fixed in \texttt{code/raw/fix\_1986.R}
  \end{itemize}
\item
  \textbf{KSIC Industry Codes} revised over times:

  \begin{itemize}
  \tightlist
  \item
    5th KSIC: 1986 - 1992
  \item
    6th/7th KSIC: 1993 - 2000
  \item
    8th KSIC: 2000 - 2008
  \item
    9th KSIC: 2004 - 2017
  \item
    10th, 11th KSIC: 2013 - 2024
  \item
    Mapping handled in \texttt{ksic\_mapping.R}
  \end{itemize}
\end{itemize}
